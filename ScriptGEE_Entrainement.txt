var bands20m = ['B5','B6','B11','B12']; // SWIR et Red edge

  }))

  .map(function(f){return f.set('class', 10);});
  
  /*
*** Certaintes classes ont été fusionnées dû à certains problèmes spectrale et spatiales,

var trainingProposition = Gazon_class.merge(RoutesParkings_class).merge(Batiments_class)
.merge(Boise_class).merge(Eau_class).merge(SurfacesMin_class)
.merge(JardinAgricUrbain_class);


//  pour l'entrainement des algo 2025
var trainingClass_proposition_1 = imageRGBNI2.sampleRegions({
    collection: trainingProposition,
    properties: ['class'],
    scale: 10, // Résolution spatiale
    geometries: true
});


// Ajouter une colonne aléatoire pour le split
var withRandom = trainingClass_proposition_1.randomColumn('random');


// 70% pour l'entraînement
var trainSet_proposition_1 = withRandom.filter(ee.Filter.lt('random', 0.7));


// 30% pour la validation
var validationSet = withRandom.filter(ee.Filter.gte('random', 0.7));


// ***--- 3. Test de séparabilité ---***

// -------------------------------------------------------------
// Fonction : Distance Jeffries–Matusita entre deux classes
// -------------------------------------------------------------

function jeffriesMatusitaFinal(image, geomA, geomB, scale) {
  
  // Moyenne pour chaque géométrie
  var meanA = ee.Array(bands.map(function(b) {
    return ee.Number(image.reduceRegion({
      reducer: ee.Reducer.mean(),
      geometry: geomA,
      scale: scale,
      maxPixels: 1e9
    }).get(b));
  }));

  var meanB = ee.Array(bands.map(function(b) {
    return ee.Number(image.reduceRegion({
      reducer: ee.Reducer.mean(),
      geometry: geomB,
      scale: scale,
      maxPixels: 1e9
    }).get(b));
  }));
  
  
 var imageArray = image.toArray();

  // Covariance pour chaque géométrie
  var covA_dict = imageArray.reduceRegion({
    reducer: ee.Reducer.centeredCovariance(),
    geometry: geomA,
    scale: scale,
    maxPixels: 1e9
  })
  

  var covB_dict = imageArray.reduceRegion({
    reducer: ee.Reducer.centeredCovariance(),
    geometry: geomB,
    scale: scale,
    maxPixels: 1e9
  })
  
  
  // Extraction sous forme d'array
  var covA = ee.Array(covA_dict.get('array'));
  var covB = ee.Array(covB_dict.get('array'));
  

  // Différence des moyennes
  var diff = meanA.subtract(meanB).reshape([bands.length, 1]);
  

  // Covariance moyenne avec régularisation
  var covMean = covA.add(covB).divide(2);
  var covMeanReg = covMean.add(ee.Array.identity(bands.length).multiply(1e-6));
  

  // Inverse de Σ
  var covInv = covMeanReg.matrixInverse();

  // Terme 1
  var term1 = diff.transpose().matrixMultiply(covInv).matrixMultiply(diff).get([0,0]).divide(8);

  // Terme 2
  var detA = covA.matrixDeterminant();
  var detB = covB.matrixDeterminant();
  var detMean = covMeanReg.matrixDeterminant();
  var term2 = detMean.divide(detA.multiply(detB).sqrt()).log().divide(2);

  // Distance JM
  var B = term1.add(term2);
  var JM = ee.Number(2).multiply(ee.Number(1).subtract(B.multiply(-1).exp()));

  return JM;
}

var JM_Gazon_Inf = jeffriesMatusitaFinal(imageRGBNI2, Gazon, Infrastructure, 10);
print('Distance Jeffries-Matusita : Gazon vs Infrastructure =', JM_Gazon_Inf);

var JM_Gazon_Boise = jeffriesMatusitaFinal(imageRGBNI2, Gazon, BoiseUrbain, 10);
print('Distance Jeffries-Matusita : Gazon vs Boise =', JM_Gazon_Boise);

var JM_Gazon_Eau = jeffriesMatusitaFinal(imageRGBNI2, Gazon, EauVisible, 10);
print('Distance Jeffries-Matusita : Gazon vs Eau =', JM_Gazon_Eau);

var JM_Gazon_SurfMinerales = jeffriesMatusitaFinal(imageRGBNI2, Gazon, SurfacesMinerales, 10);
print('Distance Jeffries-Matusita : Gazon vs Surfaces minerales =', JM_Gazon_SurfMinerales);

var JM_Gazon_JardinAgriUrbain = jeffriesMatusitaFinal(imageRGBNI2, Gazon, JardinAgriculUrbaine, 10);

print('Distance Jeffries-Matusita : Surface impermeables vs Infrastructure =', JM_SurfImp_Infr);

var JM_SurfImp_Boise = jeffriesMatusitaFinal(imageRGBNI2, SurfacesImpermeables, BoiseUrbain, 10);
print('Distance Jeffries-Matusita : Surface impermeables vs Boise =', JM_SurfImp_Boise);

var JM_SurfImp_SurfMin = jeffriesMatusitaFinal(imageRGBNI2, SurfacesImpermeables, SurfacesMinerales, 10);

// ***--- 4. Classification---***

// -- Classification par Random Forest --

var rf_2025_propo1 = ee.Classifier.smileRandomForest(500).train({
  features: trainSet_proposition_1,
  classProperty: 'class',
  inputProperties: bands
});

var image_RF_2025 = imageRGBNI2.classify(rf_2025_propo1);

var image_RF_2025_clip = image_RF_2025 // coupe sur la zone d'étude
  .clip(Etude_ULAVAL);

// Ajout de la couche
Map.addLayer(image_RF_2025_clip, {min:0, max:11, palette:
['#88d648','#bfbfbf','#de7938','#0ea300','#ffcbba','#c28097', '#c1d6bb', '#3caaff', '#ffb0a0', '#95975c', '#bf62ff', '#c0c285']}, 
'Random Forest 2025 propo 1');


// --- Smmothing ---
var seeds = ee.Algorithms.Image.Segmentation.seedGrid(4);
var snic = ee.Algorithms.Image.Segmentation.SNIC({
  image: imageRGBNI2,
  size: 4,
  compactness: 0.6,
  connectivity: 8,
  neighborhoodSize: 8,
  seeds: seeds
});


//Regroupement des pixels
var objectClass_2025_Propo1 = image_RF_2025_clip
  .addBands(snic.select('clusters'))
  .reduceConnectedComponents({
    reducer: ee.Reducer.mode(),
    labelBand: 'clusters'
  });
  
var objectClass_2025_clip = objectClass_2025_Propo1 // coupe sur la zone d'étude
  .clip(Etude_ULAVAL);

Map.addLayer(objectClass_2025_clip, {min:0, max:11, palette:
['#88d648','#bfbfbf','#de7938','#0ea300','#ffcbba','#c28097','#c1d6bb', '#3caaff', '#ffb0a0', '#95975c', '#bf62ff', '#c0c285']}, 
'Classification 2025'); // Ajout sur la carte


// ***--- 5. Matrice de confusion et validation ---***
  
// Classify the validation set to get predicted classes.
var validated  = validationSet.classify(rf_2025_propo1);

// Get the error matrix (confusion matrix) from the validated FeatureCollection.
var errorMatrix = validated.errorMatrix('class', 'classification');
print('Validation Confusion Matrix:', errorMatrix);


// Extract metrics:
print('Overall Accuracy:', errorMatrix.accuracy());
print('Kappa Coefficient:', errorMatrix.kappa());
print('Producer Accuracy:', errorMatrix.producersAccuracy());
print('Consumer Accuracy:', errorMatrix.consumersAccuracy());
